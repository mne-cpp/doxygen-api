333 template<typename T>
334 bool IOUtils::read_eigen_matrix(Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>& out, const QString& path)
335 {
336  QFile file(path);
337 
338  if(file.open(QIODevice::ReadOnly | QIODevice::Text)) {
339  //Start reading from file
340  QTextStream in(&file);
341  QList<Eigen::VectorXd> help;
342 
343  while(!in.atEnd())
344  {
345  QString line = in.readLine();
346  if(!line.contains("#")) {
347  QStringList fields = line.split(QRegExp("\\s+"));
348 
349  //Delete last element if it is a blank character
350  if(fields.at(fields.size()-1) == "")
351  fields.removeLast();
352 
353  Eigen::VectorXd x (fields.size());
354 
355  for (int j=0; j<fields.size(); j++) {
356  x(j)=fields.at(j).toDouble();
357  }
358 
359  help.append(x);
360  }
361  }
362 
363  int rows = help.size();
364  int cols = rows<=0 ? 0 : help.at(0).rows();
365 
366  out.resize(rows, cols);
367 
368  for (int i=0; i<help.length(); i++) {
369  out.row(i)=help[i].transpose();
370  }
371  } else {
372  qWarning()<<"IOUtils::read_eigen_matrix - Could not read Eigen element from file! Path does not exist!";
373  return false;
374  }
375 
376  return true;
377 }
