379 void HPIFit::findOrder(const MatrixXd& t_mat,
380  const MatrixXd& t_matProjectors,
381  FiffCoordTrans& transDevHead,
382  QVector<int>& vecFreqs,
383  QVector<double>& vecError,
384  VectorXd& vecGoF,
385  FiffDigPointSet& fittedPointSet,
386  FiffInfo::SPtr pFiffInfo)
387 {
388  // create temporary copies that are necessary to reset values that are passed to fitHpi()
389  fittedPointSet.clear();
390  transDevHead.clear();
391  vecError.fill(0);
392 
393  FiffDigPointSet fittedPointSetTemp = fittedPointSet;
394  FiffCoordTrans transDevHeadTemp = transDevHead;
395  FiffInfo::SPtr pFiffInfoTemp = pFiffInfo;
396  QVector<int> vecToOrder = vecFreqs;
397  QVector<int> vecFreqTemp(vecFreqs.size());
398  QVector<double> vecErrorTemp = vecError;
399  VectorXd vecGoFTemp = vecGoF;
400  bool bIdentity = false;
401 
402  MatrixXf matTrans = transDevHead.trans;
403  if(transDevHead.trans == MatrixXf::Identity(4,4).cast<float>()) {
404  // avoid identity since this leads to problems with this method in fitHpi.
405  // the hpi fit is robust enough to handle bad starting points
406  transDevHeadTemp.trans(3,0) = 0.000001;
407  bIdentity = true;
408  }
409 
410  // perform vecFreqs.size() hpi fits with same frequencies in each iteration
411  for(int i = 0; i < vecFreqs.size(); i++){
412  vecFreqTemp.fill(vecFreqs[i]);
413 
414  // hpi Fit
415  fitHPI(t_mat, t_matProjectors, transDevHeadTemp, vecFreqTemp, vecErrorTemp, vecGoFTemp, fittedPointSetTemp, pFiffInfoTemp);
416 
417  // get location of maximum GoF -> correct assignment of coil - frequency
418  VectorXd::Index indMax;
419  vecGoFTemp.maxCoeff(&indMax);
420  vecToOrder[indMax] = vecFreqs[i];
421 
422  // std::cout << vecGoFTemp[0] << " " << vecGoFTemp[1] << " " << vecGoFTemp[2] << " " << vecGoFTemp[3] << " " << std::endl;
423 
424  // reset values that are edidet by fitHpi()
425  fittedPointSetTemp = fittedPointSet;
426  pFiffInfoTemp = pFiffInfo;
427  transDevHeadTemp = transDevHead;
428 
429  if(bIdentity) {
430  transDevHeadTemp.trans(3,0) = 0.000001;
431  }
432  vecErrorTemp = vecError;
433  vecGoFTemp = vecGoF;
434  }
435  // check if still all frequencies are represented and update model
436  if(std::accumulate(vecFreqs.begin(), vecFreqs.end(), .0) == std::accumulate(vecToOrder.begin(), vecToOrder.end(), .0)) {
437  vecFreqs = vecToOrder;
438  } else {
439  qWarning() << "HPIFit::findOrder: frequencie ordering went wrong";
440  }
441  qInfo() << "HPIFit::findOrder: vecFreqs = " << vecFreqs;
442 }

497 Eigen::Matrix4d HPIFit::computeTransformation(Eigen::MatrixXd matNH, MatrixXd matBT)
498 {
499  MatrixXd matXdiff, matYdiff, matZdiff, matC, matQ;
500  Matrix4d matTransFinal = Matrix4d::Identity(4,4);
501  Matrix4d matRot = Matrix4d::Zero(4,4);
502  Matrix4d matTrans = Matrix4d::Identity(4,4);
503  double dMeanX,dMeanY,dMeanZ,dNormf;
504 
505  for(int i = 0; i < 15; ++i) {
506  // Calculate mean translation for all points -> centroid of both data sets
507  matXdiff = matNH.col(0) - matBT.col(0);
508  matYdiff = matNH.col(1) - matBT.col(1);
509  matZdiff = matNH.col(2) - matBT.col(2);
510 
511  dMeanX = matXdiff.mean();
512  dMeanY = matYdiff.mean();
513  dMeanZ = matZdiff.mean();
514 
515  // Apply translation -> bring both data sets to the same center location
516  for (int j = 0; j < matBT.rows(); ++j) {
517  matBT(j,0) = matBT(j,0) + dMeanX;
518  matBT(j,1) = matBT(j,1) + dMeanY;
519  matBT(j,2) = matBT(j,2) + dMeanZ;
520  }
521 
522  // Estimate rotation component
523  matC = matBT.transpose() * matNH;
524 
525  JacobiSVD< MatrixXd > svd(matC ,Eigen::ComputeThinU | ComputeThinV);
526 
527  matQ = svd.matrixU() * svd.matrixV().transpose();
528 
529  //Handle special reflection case
530  if(matQ.determinant() < 0) {
531  matQ(0,2) = matQ(0,2) * -1;
532  matQ(1,2) = matQ(1,2) * -1;
533  matQ(2,2) = matQ(2,2) * -1;
534  }
535 
536  // Apply rotation on translated points
537  matBT = matBT * matQ;
538 
539  // Calculate GOF
540  dNormf = (matNH.transpose()-matBT.transpose()).norm();
541 
542  // Store rotation part to transformation matrix
543  matRot(3,3) = 1;
544  for(int j = 0; j < 3; ++j) {
545  for(int k = 0; k < 3; ++k) {
546  matRot(j,k) = matQ(k,j);
547  }
548  }
549 
550  // Store translation part to transformation matrix
551  matTrans(0,3) = dMeanX;
552  matTrans(1,3) = dMeanY;
553  matTrans(2,3) = dMeanZ;
554 
555  // Safe rotation and translation to final matrix for next iteration step
556  // This step is safe to do since we change one of the input point sets (matBT)
557  // ToDo: Replace this for loop with a least square solution process
558  matTransFinal = matRot * matTrans * matTransFinal;
559  }
560  return matTransFinal;
561 }

684 void HPIFit::updateModel(const int iSamF,
685  const int iSamLoc,
686  int iLineF,
687  const QVector<int>& vecFreqs)
688 {
689  int iNumCoils = vecFreqs.size();
690  MatrixXd matSimsig;
691  VectorXd vecTime = VectorXd::LinSpaced(iSamLoc, 0, iSamLoc-1) *1.0/iSamF;
692 
693  if(m_bDoFastFit){
694  // Generate simulated data Matrix
695  matSimsig.conservativeResize(iSamLoc,iNumCoils*2);
696 
697  for(int i = 0; i < iNumCoils; ++i) {
698  matSimsig.col(i) = sin(2*M_PI*vecFreqs[i]*vecTime.array());
699  matSimsig.col(i+iNumCoils) = cos(2*M_PI*vecFreqs[i]*vecTime.array());
700  }
701  m_matModel = UTILSLIB::MNEMath::pinv(matSimsig);
702  return;
703 
704  } else {
705  // add linefreq + harmonics + DC part to model
706  matSimsig.conservativeResize(iSamLoc,iNumCoils*4);
707  for(int i = 0; i < iNumCoils; ++i) {
708  matSimsig.col(i) = sin(2*M_PI*vecFreqs[i]*vecTime.array());
709  matSimsig.col(i+iNumCoils) = cos(2*M_PI*vecFreqs[i]*vecTime.array());
710  matSimsig.col(i+2*iNumCoils) = sin(2*M_PI*iLineF*i*vecTime.array());
711  matSimsig.col(i+3*iNumCoils) = cos(2*M_PI*iLineF*i*vecTime.array());
712  }
713  matSimsig.col(14) = RowVectorXd::LinSpaced(iSamLoc, -0.5, 0.5);
714  matSimsig.col(15).fill(1);
715  }
716  m_matModel = UTILSLIB::MNEMath::pinv(matSimsig);
717 
718  // reorder for faster computation
719  MatrixXd matTemp = m_matModel;
720  RowVectorXi vecIndex(2*iNumCoils);
721  vecIndex << 0,4,1,5,2,6,3,7;
722  for(int i = 0; i < vecIndex.size(); ++i) {
723  matTemp.row(i) = m_matModel.row(vecIndex(i));
724  }
725  m_matModel = matTemp;
726 }
