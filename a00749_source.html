389 {
390  double *scale = NULL;
391  double *cov = NULL;
392  double *lambda = NULL;
393  float **eigen = NULL;
394  double **data1 = NULL;
395  double **data2 = NULL;

399  int res = FAIL;
400 
401  if (c->cov_diag)
402  return OK;

404  qCritical("Channels not classified. Rank cannot be determined.");
405  return FAIL;
406  }
407  magscale = gradscale = eegscale = 0.0;
408  nmag = ngrad = neeg = 0;
409  for (k = 0; k < c->ncov; k++) {
410  if (c->ch_class[k] == MNE_COV_CH_MEG_MAG) {
411  magscale += c->cov[mne_lt_packed_index(k,k)]; nmag++;
412  }
413  else if (c->ch_class[k] == MNE_COV_CH_MEG_GRAD) {
414  gradscale += c->cov[mne_lt_packed_index(k,k)]; ngrad++;
415  }
416  else if (c->ch_class[k] == MNE_COV_CH_EEG) {
417  eegscale += c->cov[mne_lt_packed_index(k,k)]; neeg++;
418  }
419 #ifdef DEBUG
420  fprintf(stdout,"%d ",c->ch_class[k]);
421 #endif
422  }
423 #ifdef DEBUG
424  fprintf(stdout,"\n");
425 #endif
426  if (nmag > 0)
427  magscale = magscale > 0.0 ? sqrt(nmag/magscale) : 0.0;
428  if (ngrad > 0)
429  gradscale = gradscale > 0.0 ? sqrt(ngrad/gradscale) : 0.0;
430  if (neeg > 0)
431  eegscale = eegscale > 0.0 ? sqrt(neeg/eegscale) : 0.0;
432 #ifdef DEBUG
433  fprintf(stdout,"%d %g\n",nmag,magscale);
434  fprintf(stdout,"%d %g\n",ngrad,gradscale);
435  fprintf(stdout,"%d %g\n",neeg,eegscale);
436 #endif
437  scale = MALLOC_30(c->ncov,double);
438  for (k = 0; k < c->ncov; k++) {
439  if (c->ch_class[k] == MNE_COV_CH_MEG_MAG)
440  scale[k] = magscale;
441  else if (c->ch_class[k] == MNE_COV_CH_MEG_GRAD)
442  scale[k] = gradscale;
443  else if (c->ch_class[k] == MNE_COV_CH_EEG)
444  scale[k] = eegscale;
445  else
446  scale[k] = 1.0;
447  }
448  cov = MALLOC_30(c->ncov*(c->ncov+1)/2.0,double);
449  lambda = MALLOC_30(c->ncov,double);
450  eigen = ALLOC_CMATRIX_30(c->ncov,c->ncov);
451  for (j = 0; j < c->ncov; j++)
452  for (k = 0; k <= j; k++)
453  cov[mne_lt_packed_index(j,k)] = c->cov[mne_lt_packed_index(j,k)]*scale[j]*scale[k];
454  if (mne_decompose_eigen(cov,lambda,eigen,c->ncov) == 0) {
455 #ifdef DEBUG
456  for (k = 0; k < c->ncov; k++)
457  fprintf(stdout,"%g ",lambda[k]/lambda[c->ncov-1]);
458  fprintf(stdout,"\n");
459 #endif
460  nok = 0;
461  for (k = c->ncov-1; k >= 0; k--) {
462  if (lambda[k] >= rank_threshold*lambda[c->ncov-1])
463  nok++;
464  else
465  break;
466  }
467  printf("\n\tEstimated covariance matrix rank = %d (%g)\n",nok,lambda[c->ncov-nok]/lambda[c->ncov-1]);
468  if (use_rank > 0 && use_rank < nok) {
469  nok = use_rank;
470  fprintf(stderr,"\tUser-selected covariance matrix rank = %d (%g)\n",nok,lambda[c->ncov-nok]/lambda[c->ncov-1]);
471  }
472  /*
473  * Put it back together
474  */
475  for (j = 0; j < c->ncov-nok; j++)
476  lambda[j] = 0.0;
477  data1 = ALLOC_DCMATRIX_30(c->ncov,c->ncov);
478  for (j = 0; j < c->ncov; j++) {
479 #ifdef DEBUG
480  mne_print_vector(stdout,NULL,eigen[j],c->ncov);
481 #endif
482  for (k = 0; k < c->ncov; k++)
483  data1[j][k] = sqrt(lambda[j])*eigen[j][k];
484  }
485  data2 = mne_dmatt_dmat_mult2(data1,data1,c->ncov,c->ncov,c->ncov);
486 #ifdef DEBUG
487  printf(">>>\n");
488  for (j = 0; j < c->ncov; j++)
489  mne_print_dvector(stdout,NULL,data2[j],c->ncov);
490  printf(">>>\n");
491 #endif
492  /*
493  * Scale back
494  */
495  for (k = 0; k < c->ncov; k++)
496  if (scale[k] > 0.0)
497  scale[k] = 1.0/scale[k];
498  for (j = 0; j < c->ncov; j++)
499  for (k = 0; k <= j; k++)
500  if (c->cov[mne_lt_packed_index(j,k)] != 0.0)
501  c->cov[mne_lt_packed_index(j,k)] = scale[j]*scale[k]*data2[j][k];
502  res = nok;
503  }
504  FREE_30(cov);
505  FREE_30(lambda);
506  FREE_CMATRIX_30(eigen);
507  FREE_DCMATRIX_30(data1);
508  FREE_DCMATRIX_30(data2);
509  return res;
510 }

515 /*
516  * Do the eigenvalue decomposition
517  */
518 {
519  int np,k,p,rank;
520  float rank_threshold = 1e-6;
521 
522  if (p_small < 0)
523  p_small = 1.0;
524 
525  if (!c)
526  return OK;
527  if (c->cov_diag)
528  return mne_add_inv_cov(c);
529  if (c->lambda && c->eigen) {
530  fprintf(stderr,"\n\tEigenvalue decomposition had been precomputed.\n");
531  c->nzero = 0;
532  for (k = 0; k < c->ncov; k++, c->nzero++)
533  if (c->lambda[k] > 0)
534  break;
535  }
536  else {
537  FREE_30(c->lambda); c->lambda = NULL;
538  FREE_CMATRIX_30(c->eigen); c->eigen = NULL;
539 
540  if ((rank = condition_cov(c,rank_threshold,use_rank)) < 0)
541  return FAIL;
542 
543  np = c->ncov*(c->ncov+1)/2;
544  c->lambda = MALLOC_30(c->ncov,double);
545  c->eigen = ALLOC_CMATRIX_30(c->ncov,c->ncov);
546  if (mne_decompose_eigen (c->cov,c->lambda,c->eigen,c->ncov) != 0)
547  goto bad;
548  c->nzero = c->ncov - rank;
549  for (k = 0; k < c->nzero; k++)
550  c->lambda[k] = 0.0;
551  /*
552  * Find which eigenvectors correspond to EEG/MEG
553  */
554  {
555  float meglike,eeglike;
556  int nmeg,neeg;
557 
558  nmeg = neeg = 0;
559  for (k = c->nzero; k < c->ncov; k++) {
560  meglike = eeglike = 0.0;
561  for (p = 0; p < c->ncov; p++) {
562  if (c->ch_class[p] == MNE_COV_CH_EEG)
563  eeglike += std::fabs(c->eigen[k][p]);
564  else if (c->ch_class[p] == MNE_COV_CH_MEG_MAG || c->ch_class[p] == MNE_COV_CH_MEG_GRAD)
565  meglike += std::fabs(c->eigen[k][p]);
566  }
567  if (meglike > eeglike)
568  nmeg++;
569  else
570  neeg++;
571  }
572  printf("\t%d MEG and %d EEG-like channels remain in the whitened data\n",nmeg,neeg);
573  }
574  }
575  return mne_add_inv_cov(c);
576 
577 bad : {
578  FREE_30(c->lambda); c->lambda = NULL;
579  FREE_CMATRIX_30(c->eigen); c->eigen = NULL;
580  return FAIL;
581  }
582 }
