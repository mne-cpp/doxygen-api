399 {
400  m_pUi->m_tableWidget_Frequencies->clear();
401  m_pUi->m_tableWidget_Frequencies->setRowCount(0);
402  m_pUi->m_tableWidget_Frequencies->setHorizontalHeaderItem(0, new QTableWidgetItem("#Coil"));
403  m_pUi->m_tableWidget_Frequencies->setHorizontalHeaderItem(1, new QTableWidgetItem("Frequency (Hz)"));
404 
405  m_pUi->m_tableWidget_errors->clear();
406  m_pUi->m_tableWidget_errors->setRowCount(0);
407  m_pUi->m_tableWidget_errors->setHorizontalHeaderItem(0, new QTableWidgetItem("#Coil"));
408  m_pUi->m_tableWidget_errors->setHorizontalHeaderItem(1, new QTableWidgetItem("Error"));
409 
410  QFile t_fileDig(fileName);
411  FiffDigPointSet t_digSet(t_fileDig);
412 
413  QList<FiffDigPoint> lDigPoints;
414 
415  qint16 numHPI = 0;
416  qint16 numFiducials = 0;
417  qint16 numEEG = 0;
418  qint16 numAdditional = 0;
419 
420  for(int i = 0; i < t_digSet.size(); ++i) {
421  switch(t_digSet[i].kind) {
422  case FIFFV_POINT_HPI: {
423  // Add column 0 in freq table widget
424  m_pUi->m_tableWidget_Frequencies->insertRow(m_pUi->m_tableWidget_Frequencies->rowCount());
425  QTableWidgetItem* pTableItemA = new QTableWidgetItem(QString::number(m_pUi->m_tableWidget_Frequencies->rowCount()));
426  pTableItemA->setFlags(Qt::ItemIsEnabled);
427  m_pUi->m_tableWidget_Frequencies->setItem(m_pUi->m_tableWidget_Frequencies->rowCount()-1,
428  0,
429  pTableItemA);
430 
431  // Add column 1 in freq table widget
432  if(m_vCoilFreqs.size() > numHPI) {
433  m_pUi->m_tableWidget_Frequencies->setItem(m_pUi->m_tableWidget_Frequencies->rowCount()-1,
434  1,
435  new QTableWidgetItem(QString::number(m_vCoilFreqs.at(numHPI))));
436  } else {
437  m_pUi->m_tableWidget_Frequencies->setItem(m_pUi->m_tableWidget_Frequencies->rowCount()-1,
438  1,
439  new QTableWidgetItem("none"));
440  m_vCoilFreqs.append(-1);
441  }
442 
443  // Add column 0 in error table widget
444  m_pUi->m_tableWidget_errors->insertRow(m_pUi->m_tableWidget_errors->rowCount());
445  QTableWidgetItem* pTableItemB = new QTableWidgetItem(QString::number(m_pUi->m_tableWidget_Frequencies->rowCount()));
446  pTableItemB->setFlags(Qt::ItemIsEnabled);
447  m_pUi->m_tableWidget_errors->setItem(m_pUi->m_tableWidget_errors->rowCount()-1,
448  0,
449  pTableItemB);
450 
451  // Add column 1 in error table widget
452  QTableWidgetItem* pTableItemC = new QTableWidgetItem("0mm");
453  pTableItemC->setFlags(Qt::ItemIsEnabled);
454  m_pUi->m_tableWidget_errors->setItem(m_pUi->m_tableWidget_errors->rowCount()-1,
455  1,
456  pTableItemC);
457 
458  lDigPoints.append(t_digSet[i]);
459  numHPI++;
460  break;
461  }
462 
463  case FIFFV_POINT_CARDINAL:
464  lDigPoints.append(t_digSet[i]);
465  numFiducials++;
466  break;
467 
468  case FIFFV_POINT_EEG:

470  numEEG++;
471  break;
472 
473  case FIFFV_POINT_EXTRA:
474  lDigPoints.append(t_digSet[i]);
475  numAdditional++;
476  break;
477  }
478  }
479 
480  //Set loaded number of digitizers
481  m_pUi->m_label_numberLoadedCoils->setNum(numHPI);
482  m_pUi->m_label_numberLoadedFiducials->setNum(numFiducials);
483  m_pUi->m_label_numberLoadedEEG->setNum(numEEG);
484  m_pUi->m_label_numberLoadedAdditional->setNum(numAdditional);
485

487  m_vCoilFreqs.resize(numHPI);
488  emit coilFrequenciesChanged(m_vCoilFreqs);
489 
490  return lDigPoints;
491 }
