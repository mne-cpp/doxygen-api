310 void View3D::createCoordSystem(Qt3DCore::QEntity* parent)
311 {
312  m_pCoordSysEntity = QSharedPointer<Qt3DCore::QEntity>::create(parent);
313 
314  //create geometry
315  QSharedPointer<Qt3DExtras::QCylinderGeometry> pAxis = QSharedPointer<Qt3DExtras::QCylinderGeometry>::create();
316  pAxis->setRadius(0.001f);
317  pAxis->setLength(30);
318  pAxis->setRings(100);
319  pAxis->setSlices(20);
320 
321  //create mesh
322  GeometryMultiplier *pCoordSysMesh = new GeometryMultiplier(pAxis);
323  QVector<QColor> vColors;
324  vColors.reserve(3);
325  QVector<QMatrix4x4> vTransforms;
326  vTransforms.reserve(3);
327  QMatrix4x4 transformMat;
328 
329  // Y - red
330  transformMat.setToIdentity();
331  vTransforms.push_back(transformMat);
332  vColors.push_back(QColor(255, 0, 0));
333 
334  // X - blue
335  transformMat.setToIdentity();
336  transformMat.rotate(90.0f, QVector3D(0,0,1));
337  vTransforms.push_back(transformMat);
338  vColors.push_back(QColor(0, 0, 255));
339 
340  // Z - green
341  transformMat.setToIdentity();
342  transformMat.rotate(90.0f, QVector3D(1,0,0));
343  vTransforms.push_back(transformMat);
344  vColors.push_back(QColor(0, 255, 0));
345 
346  //Set transforms and colors
347  pCoordSysMesh->setTransforms(vTransforms);
348  pCoordSysMesh->setColors(vColors);
349 
350  //Add material
351  GeometryMultiplierMaterial* pCoordSysMaterial = new GeometryMultiplierMaterial;
352  pCoordSysMaterial->setAmbient(QColor(0,0,0));
353  pCoordSysMaterial->setAlpha(1.0f);
